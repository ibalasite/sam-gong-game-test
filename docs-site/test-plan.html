3. Integration Tests — Colyseus Room

// 使用 Colyseus Test SDK
import { ColyseusTestServer, boot } from "@colyseus/testing";

describe("SamGongRoom", () => {
  let server: ColyseusTestServer;

  beforeAll(async () => {
    server = await boot(app);
  });

  it("IT-ROOM-001: 2 players can join and start game", async () => {
    const room = await server.createRoom("sam_gong", { minBet: 10 });
    const client1 = await server.connectTo(room);
    const client2 = await server.connectTo(room);

    await client1.send("start_game", {});
    await room.waitForMessage("game_started");

    expect(room.state.phase).toBe("banker_selection");
  });
});

4. E2E Tests — Playwright

// playwright/specs/full-game.spec.ts
test("E2E-001: 完整遊戲流程（2 玩家）", async ({ browser }) => {
  const ctx1 = await browser.newContext();
  const ctx2 = await browser.newContext();
  const page1 = await ctx1.newPage();
  const page2 = await ctx2.newPage();

  await page1.goto("http://localhost:8080");
  // 創建房間
  await page1.click("[data-testid='create-room']");
  const roomCode = await page1.locator("[data-testid='room-code']").textContent();

  // 加入房間
  await page2.goto("http://localhost:8080");
  await page2.fill("[data-testid='room-code-input']", roomCode!);
  await page2.click("[data-testid='join-room']");

  // 開始遊戲
  await page1.click("[data-testid='start-game']");
  await expect(page1.locator("[data-testid='phase']")).toHaveText("betting");
});

import ws from 'k6/ws';

export const options = {
  scenarios: {
