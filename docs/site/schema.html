SCHEMA — 資料庫 Schema 文件 v1.2

1. Design Principles

  1. ACID 事務優先：籌碼結算使用 SERIALIZABLE 隔離等級，確保籌碼守恆（誤差容忍 = 0）
  2. 財務記錄保留 7 年：chip_transactions 按月分區（pg_partman），帳號刪除後 user_id 匿名化
  3. 不可變記錄：chip_transactions 不支援 UPDATE/DELETE（僅 INSERT）
  4. 讀寫分離：排行榜、玩家資料讀取走 Read Replica；寫入走 Primary
  5. 索引策略：優先支援高頻查詢（結算、排行榜、Rate Limit）

3. Table Definitions

3.1 users

CREATE TABLE users (
    id                      UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    display_name            VARCHAR(24) NOT NULL,
    phone_hash              VARCHAR(64),          -- SHA-256(phone), never plaintext
    oauth_provider          VARCHAR(20),           -- 'google'|'facebook'|null
    oauth_id                VARCHAR(128),
    age_verified            BOOLEAN NOT NULL DEFAULT FALSE,
    birth_year              SMALLINT,
    is_minor                BOOLEAN NOT NULL DEFAULT FALSE,
    is_banned               BOOLEAN NOT NULL DEFAULT FALSE,
    ban_reason              TEXT,
    chip_balance            BIGINT NOT NULL DEFAULT 100000,
    created_at              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    daily_chip_claimed_at   DATE,
    daily_rescue_claimed_at DATE,
    tutorial_completed      BOOLEAN NOT NULL DEFAULT FALSE,
    avatar_url              TEXT,
    music_volume            SMALLINT NOT NULL DEFAULT 70
                              CHECK (music_volume BETWEEN 0 AND 100),
    sfx_volume              SMALLINT NOT NULL DEFAULT 80
                              CHECK (sfx_volume BETWEEN 0 AND 100),
    vibration               BOOLEAN NOT NULL DEFAULT TRUE,
    show_in_leaderboard     BOOLEAN NOT NULL DEFAULT TRUE,
    daily_play_seconds      INT NOT NULL DEFAULT 0,
    session_play_seconds    INT NOT NULL DEFAULT 0,
    last_login_at           TIMESTAMPTZ,
    UNIQUE (oauth_provider, oauth_id),
    CONSTRAINT chip_balance_non_negative CHECK (chip_balance >= 0)
);

CREATE INDEX idx_users_chip_balance ON users(chip_balance DESC);
CREATE INDEX idx_users_is_banned ON users(is_banned) WHERE is_banned = TRUE;

3.2 game_sessions

CREATE TABLE game_sessions (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    room_id             UUID NOT NULL REFERENCES game_rooms(id),
    round_number        INT NOT NULL,
    banker_id           UUID REFERENCES users(id) ON DELETE SET NULL,
    banker_seat_index   INT NOT NULL,
    banker_bet_amount   BIGINT NOT NULL CHECK (banker_bet_amount >= 0),
    rake_amount         BIGINT NOT NULL DEFAULT 0 CHECK (rake_amount >= 0),
    pot_amount          BIGINT NOT NULL DEFAULT 0,
    banker_insolvent    BOOLEAN NOT NULL DEFAULT FALSE,
    all_fold            BOOLEAN NOT NULL DEFAULT FALSE,
    settlement_payload  JSONB NOT NULL,          -- full settlement snapshot
    started_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    ended_at            TIMESTAMPTZ
);

CREATE INDEX idx_sessions_room_id ON game_sessions(room_id);
CREATE INDEX idx_sessions_started_at ON game_sessions(started_at DESC);

3.3 chip_transactions

-- Partitioned by month for 7-year retention
CREATE TABLE chip_transactions (
    id                  UUID NOT NULL DEFAULT uuid_generate_v4(),
    user_id             UUID REFERENCES users(id) ON DELETE SET NULL,
    game_session_id     UUID REFERENCES game_sessions(id) ON DELETE SET NULL,
    tx_type             VARCHAR(30) NOT NULL
                          CHECK (tx_type IN (
                            'win', 'loss', 'rake', 'daily_chip',
                            'rescue_chip', 'iap_purchase', 'ad_reward',
                            'daily_task_reward', 'admin_adjust',
                            'tutorial_initial'
                          )),
    amount              BIGINT NOT NULL,         -- positive=credit, negative=debit
    balance_before      BIGINT NOT NULL,
    balance_after       BIGINT NOT NULL,
    metadata            JSONB,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
) PARTITION BY RANGE (created_at);

-- Example: monthly partition
CREATE TABLE chip_transactions_2026_04
    PARTITION OF chip_transactions
    FOR VALUES FROM ('2026-04-01') TO ('2026-05-01');

CREATE INDEX idx_tx_user_id ON chip_transactions(user_id, created_at DESC);
CREATE INDEX idx_tx_session_id ON chip_transactions(game_session_id)
    WHERE game_session_id IS NOT NULL;

4. Settlement Transaction Pattern

-- SERIALIZABLE transaction for settlement
BEGIN TRANSACTION ISOLATION LEVEL SERIALIZABLE;

-- 1. Deduct from losers
UPDATE users SET chip_balance = chip_balance + :net_chips
WHERE id = :player_id;

-- 2. Record each transaction
INSERT INTO chip_transactions (user_id, game_session_id, tx_type, amount,
                                balance_before, balance_after)
VALUES (:player_id, :session_id, 'win'|'loss', :amount,
        :balance_before, :balance_after);

-- 3. Rake record (user_id = NULL = rake account)
INSERT INTO chip_transactions (user_id, game_session_id, tx_type, amount,
                                balance_before, balance_after)
VALUES (NULL, :session_id, 'rake', :rake_amount, 0, 0);

-- 4. Verify conservation (application-level check before commit)
-- sum(net_chips) + rake === 0

COMMIT;
