1. Executive Summary

三公遊戲（Sam Gong 3-Card Poker）是一款 Server-Authoritative 即時多人紙牌遊戲。所有遊戲邏輯（洗牌、發牌、比牌、結算）完全由 Server 執行，Client（Cocos Creator 3.8.x）僅負責狀態渲染與使用者互動收集，實現「啞渲染器（dumb renderer）」架構。

關鍵架構決策

  1. Server-Authoritative 架構：Colyseus 0.15.x（Node.js 22.x + TypeScript 5.4.x）作為 Game Server，透過 WebSocket 即時同步 Room State；Client 不包含任何遊戲邏輯。
  2. 資料持久層：PostgreSQL 16.x（玩家資料、交易記錄、KYC）；Redis 7.x（快取、Session、Rate Limit、Leaderboard Sorted Set）
  3. 身份驗證：JWT RS256/ES256；Access Token ≤ 1 小時，Refresh Token ≤ 7 天；帳號封鎖後 ≤ 60 秒失效
  4. 部署：Kubernetes（k8s）叢集，Colyseus 水平擴展配合 @colyseus/redis-presence；GitHub Actions CI/CD
  5. 台灣法規合規：虛擬籌碼不可兌換、OTP 年齡驗證、防沉迷計時器、KYC

3.1 SamGongRoom — Room State Schema

// SamGongRoomState — Colyseus Schema
import { Schema, MapSchema, ArraySchema, type } from "@colyseus/schema";

export class CardSchema extends Schema {
  @type("string") suit: string;   // 'spades'|'hearts'|'diamonds'|'clubs'
  @type("number") rank: number;   // 1(A),2-10,11(J),12(Q),13(K)
}

export class PlayerState extends Schema {
  @type("string") player_id: string;
  @type("string") display_name: string;
  @type("number") chip_balance: number;  // Open Information
  @type("number") seat_index: number;    // 0-based
  @type("boolean") is_banker: boolean;
  @type("boolean") is_connected: boolean;
  @type("string") action: string;        // 'waiting'|'call'|'fold'
  @type("number") bet_amount: number;
  // session_id: Server-only, NOT decorated with @type
  session_id: string;
}

export class SamGongRoomState extends Schema {
  @type("string") phase: string;         // 'waiting'|'dealing'|'banker-bet'|'player-bet'|'showdown'|'settled'
  @type({ map: PlayerState }) players = new MapSchema();
  @type("string") current_banker_id: string;
  @type("number") banker_bet_amount: number;
  @type("number") pot_amount: number;
  @type("string") current_player_turn_id: string;
  @type("number") action_deadline: number;  // Unix timestamp ms
  @type("number") round_number: number;
  @type("string") room_code: string;
  @type("string") tier: string;
}

3.3 HandEvaluator 關鍵邏輯

// HandEvaluator.ts
const SUIT_RANK = { spades: 4, hearts: 3, diamonds: 2, clubs: 1 };
const FACE_VALUE = { 1: 1, 11: 0, 12: 0, 13: 0 }; // A=1, J=Q=K=0

export function calculateHandValue(cards: CardSchema[]): number {
  const sum = cards.reduce((acc, card) => {
    const val = card.rank >= 10 ? 0 : (card.rank === 1 ? 1 : card.rank);
    return acc + val;
  }, 0);
  return sum % 10;
}

export function isSamGong(cards: CardSchema[]): boolean {
  return cards.every(c => c.rank === 1 || c.rank >= 10);
}

// Returns positive if hand1 > hand2, negative if hand1 < hand2, 0 if tie
export function compareHands(hand1: CardSchema[], hand2: CardSchema[]): number {
  const sg1 = isSamGong(hand1), sg2 = isSamGong(hand2);
  if (sg1 !== sg2) return sg1 ? 1 : -1;
  const v1 = calculateHandValue(hand1), v2 = calculateHandValue(hand2);
  if (v1 !== v2) return v1 - v2;
  // Tiebreak by highest suit
  const maxSuit = (h: CardSchema[]) => Math.max(...h.map(c => SUIT_RANK[c.suit]));
  return maxSuit(hand1) - maxSuit(hand2);
}

3.4 SettlementEngine — Rake Formula

// Rake = floor(pot * 0.05), minimum 1
// pot = sum of all callers' bet amounts (not folded)
// Conservation: sum(net_chips) + rake === 0
const rake = Math.max(1, Math.floor(pot * 0.05));
// Validation
const conservation = settlements.reduce((a, s) => a + s.net_chips, 0) + rake;
if (conservation !== 0) throw new ChipConservationError(conservation);

4. Security Architecture

4.1 Multi-Layer Defense

層次機制目的
L1: EdgeCloudFlare DDoS + GeoIP網路層攻擊防護
L2: IngressNGINX Rate Limit（連線數、請求頻率）過載保護
L3: APIRedis Token Bucket（rl:auth:{ip} ≤ 30/min）暴力破解防護
L4: WSColyseus onMessage 驗證（phase、is_banker、餘額）非法訊息過濾
L5: DBSERIALIZABLE + parameterized queries籌碼守恆 + SQL Injection 防護

4.2 JWT 黑名單機制（封號即時踢出）
